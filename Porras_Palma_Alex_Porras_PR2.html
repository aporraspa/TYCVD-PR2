def fetch_genres(title):
    params = {
        'key': API_KEY,
        'search': title,
        'page_size': 1
    }
    try:
        response = requests.get(BASE_URL, params=params)
        response.raise_for_status()
        data = response.json()
        if data['results']:
            genres = [genre['name'] for genre in data['results'][0].get('genres', [])]
            return ', '.join(genres)
    except Exception as e:
        print(f"Error fetching genres for '{title}': {e}")
    return 'Genre not found'
